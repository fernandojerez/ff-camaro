package ff.test;

import java.io.File;
import java.nio.file.Path;

import ff.crl.test.FFTestEvent;
import ff.crl.test.FFTestSuite;
import ff.dart.charger.test.FFDartTestEvent;

public class FFDartTestSuite extends FFTestSuite {

	@Override
	protected FFTestEvent getTestEvent() {
		String projectName = "#project.name#";
		String build_dir = System.getenv("FF_BUILD_DIR");
		String resolve = "";
		if(build_dir != null) {
			resolve = build_dir + "/" + projectName + "/";
		}
		return new FFDartTestEvent(new File(resolve + "build/classes/ff_dart/test").toPath());
	}

	@Override
	protected Path getTestPath() {
		return new File("src/test/ff").toPath();
	}

}
